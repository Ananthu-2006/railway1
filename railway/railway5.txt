// ========== MySQL BACKEND CONFIGURATION ==========
// The backend PHP API endpoint (api.php) handles all MySQL operations.
// Database setup script is provided in comments inside PHP file.
const API_URL = 'api.php';

let activeEmpId = null;
let activeDept = null;
let barChart = null;
let departmentsData = [];
let deptEdMap = {};  // { deptno: [edcodes] }
let fullEdCodes = [];

// ========== Helper: fetch from API ==========
async function apiRequest(action, payload = {}) {
  try {
    const res = await fetch(API_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ action, ...payload })
    });
    const data = await res.json();
    if (!data.success) throw new Error(data.error || 'API error');
    return data;
  } catch (err) {
    console.error(`API ${action} error:`, err);
    throw err;
  }
}

// ========== Load all master data from MySQL ==========
async function loadMasterData() {
  try {
    const [deptRes, edRes, mappingRes] = await Promise.all([
      apiRequest('getDepartments'),
      apiRequest('getEdCodes'),
      apiRequest('getDeptEdMapping')
    ]);
    departmentsData = deptRes.data;
    fullEdCodes = edRes.data;
    // build mapping: deptEdMap[deptno] = array of ED code objects
    const mapping = mappingRes.data; // [{deptno, edcode}]
    deptEdMap = {};
    departmentsData.forEach(d => { deptEdMap[d.deptno] = []; });
    mapping.forEach(m => {
      const edObj = fullEdCodes.find(e => e.code === m.edcode);
      if (edObj && deptEdMap[m.deptno]) deptEdMap[m.deptno].push(edObj);
    });
    document.getElementById('db-status').innerHTML = '✅ MySQL database connected | Data loaded from railway_ed_system';
    setTimeout(() => {
      const statusDiv = document.getElementById('db-status');
      if(statusDiv) statusDiv.style.opacity = '0.7';
    }, 3000);
    return true;
  } catch (err) {
    document.getElementById('db-status').innerHTML = '⚠️ MySQL connection failed. Using fallback demo data (setup required). Check api.php configuration.';
    console.error('MySQL load error:', err);
    // Fallback to embedded demo data (for standalone testing without DB)
    loadFallbackData();
    return false;
  }
}

// ========== LOGIN (credentials hardcoded or from DB) ==========
async function doLogin() {

  const pw = document.getElementById('password').value;

  try {
    const res = await apiRequest('login', { empid, password: pw });
    if (res.success && res.valid) {
      err.classList.remove('show');

      <div class="dept-card-badge">${d.arpan}</div>
      <div class="dept-count">${d.deptno}</div>
      <h4>${d.deptdesc}</h4>
      <p>${edCount} ED codes</p>
    `;
    card.addEventListener('click', () => selectDept(d.deptno));
    grid.appendChild(card);
  });
}

function selectDept(deptno) {
  activeDept = activeDept === deptno ? null : deptno;
  document.querySelectorAll('.dept-card').forEach(c => {
    c.classList.toggle('active', c.dataset.deptno === activeDept);
  });
  renderEdTable(activeDept);
  updateCharts(activeDept);
}

function renderEdTable(deptno) {
  const tbody = document.getElementById('ed-tbody');
  const title = document.getElementById('ed-table-title');
  const desc = document.getElementById('ed-table-desc');
  let codes = deptno ? (deptEdMap[deptno] || []) : fullEdCodes;
  const dept = departmentsData.find(d => d.deptno === deptno);
  title.textContent = dept ? dept.deptdesc + ' — ED Codes' : 'All ED Codes';
  desc.textContent = `Showing ${codes.length} earning/deduction code${codes.length !== 1 ? 's' : ''}${dept ? ' for ' + dept.deptdesc : ' (MySQL data)'}`;
  tbody.innerHTML = codes.map(e => {
    const isDeduct = e.code.startsWith('D') || e.long.toLowerCase().includes('deduction') || e.long.toLowerCase().includes('recovery');
    const type = isDeduct ? 'Deduction' : 'Earning';
    const typeBadge = isDeduct ? 'badge-d' : 'badge-e';
    return `<tr><td><span class="badge-code">${e.code}</span></td><td style="font-weight:500;font-size:12px;">${e.short}</td><td style="font-size:12px;color:#5f5e5a;">${e.long}</td><td><span class="badge-code ${typeBadge}">${type}</span></td></tr>`;
  }).join('');
}

function renderCharts() {
  const barLabels = departmentsData.map(d => d.arpan);
  const barData = departmentsData.map(d => (deptEdMap[d.deptno] || []).length);
  const ctx = document.getElementById('barChart').getContext('2d');
  if(barChart) barChart.destroy();
  barChart = new Chart(ctx, {
    type: 'bar',
    data: { labels: barLabels, datasets: [{ label: 'ED codes', data: barData, backgroundColor: deptColors.map(c => c + 'cc'), borderColor: deptColors, borderWidth: 1.5, borderRadius: 5 }] },
    options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { x: { grid: { display: false }, ticks: { font: { size: 11 }, autoSkip: false, maxRotation: 45 } }, y: { grid: { color: '#f1efe8' }, ticks: { font: { size: 11 } } } } }
  });
}

function updateCharts(deptno) {
  if (!barChart) return;
  barChart.data.datasets[0].backgroundColor = departmentsData.map((d, i) => {
    const base = deptColors[i % deptColors.length];
    if (!deptno) return base + 'cc';
    return d.deptno === deptno ? base : base + '44';
  });
  barChart.update();
}

function renderFlowchart() {
  const wrap = document.getElementById('flowchart-svg-wrap');
  const svgW = 1100, svgH = 520;
  const rootX = 520, rootY = 50, rootW = 160, rootH = 44;
  let svg = `<svg viewBox="0 0 ${svgW} ${svgH}" xmlns="http://www.w3.org/2000/svg" style="width:100%;max-width:${svgW}px;display:block;"><defs><marker id="arr" markerWidth="8" markerHeight="8" refX="6" refY="3" orient="auto"><path d="M0,0 L0,6 L8,3 z" fill="#9a9890"/></marker></defs>`;
  svg += `<rect x="${rootX - rootW/2}" y="${rootY}" width="${rootW}" height="${rootH}" rx="8" fill="#0f2346"/><text x="${rootX}" y="${rootY + 26}" text-anchor="middle" font-size="13" font-weight="600" fill="#ffffff">Indian Railways</text>`;
  const midY = rootY + rootH + 50;
  svg += `<line x1="${rootX}" y1="${rootY + rootH}" x2="${rootX}" y2="${midY - 10}" stroke="#9a9890" stroke-width="1.5" marker-end="url(#arr)"/>`;
  svg += `<rect x="${rootX - 75}" y="${midY - 10}" width="150" height="36" rx="6" fill="#e8a020"/><text x="${rootX}" y="${midY + 12}" text-anchor="middle" font-size="12" font-weight="600" fill="#0f2346">13 Departments</text>`;
  const row1Depts = departmentsData.slice(0, 7), row2Depts = departmentsData.slice(7);
  const row1Y = midY + 70, row2Y = row1Y + 90;
  const gap1 = svgW / (row1Depts.length + 1), gap2 = svgW / (row2Depts.length + 1);
  row1Depts.forEach((d, i) => {
    const cx = gap1 * (i + 1), cy = row1Y;
    const color = deptColors[i % deptColors.length];

    const edHtml = encodeURIComponent(JSON.stringify({ dept: d.deptdesc, codes: edList }));

    svg += `<g class="flow-node" data-ed="${edHtml}" onmouseenter="showFlowTip(event,this)" onmouseleave="hideTip()"><rect x="${cx - nodeW/2}" y="${cy - nodeH/2}" width="130" height="38" rx="7" fill="${color}" opacity="0.92"/><text x="${cx}" y="${cy - 4}" text-anchor="middle" font-size="11" font-weight="600" fill="#ffffff">${d.deptdesc}</text><text x="${cx}" y="${cy + 10}" text-anchor="middle" font-size="10" fill="rgba(255,255,255,0.7)">${d.arpan} · ${(deptEdMap[d.deptno]||[]).length} codes</text></g>`;
  });
  row2Depts.forEach((d, i) => {
    const cx = gap2 * (i + 1), cy = row2Y;
    const color = deptColors[(i + 7) % deptColors.length];
    const edList = (deptEdMap[d.deptno] || []).slice(0, 6);
    const edHtml = encodeURIComponent(JSON.stringify({ dept: d.deptdesc, codes: edList }));
    svg += `<line x1="${rootX}" y1="${midY + 26}" x2="${cx}" y2="${cy}" stroke="#9a9890" stroke-width="1" stroke-dasharray="4,3"/>`;
    svg += `<g class="flow-node" data-ed="${edHtml}" onmouseenter="showFlowTip(event,this)" onmouseleave="hideTip()"><rect x="${cx - nodeW/2}" y="${cy - nodeH/2}" width="130" height="38" rx="7" fill="${color}" opacity="0.92"/><text x="${cx}" y="${cy - 4}" text-anchor="middle" font-size="11" font-weight="600" fill="#ffffff">${d.deptdesc}</text><text x="${cx}" y="${cy + 10}" text-anchor="middle" font-size="10" fill="rgba(255,255,255,0.7)">${d.arpan} · ${(deptEdMap[d.deptno]||[]).length} codes</text></g>`;
  });
  svg += `<rect x="30" y="${row2Y + 65}" width="140" height="36" rx="6" fill="#e1f5ee"/><text x="100" y="${row2Y + 80}" text-anchor="middle" font-size="11" font-weight="600" fill="#0f6e56">Earning Codes</text><text x="100" y="${row2Y + 94}" text-anchor="middle" font-size="10" fill="#1d9e75">Prefix E</text>`;
  svg += `<rect x="200" y="${row2Y + 65}" width="140" height="36" rx="6" fill="#faece7"/><text x="270" y="${row2Y + 80}" text-anchor="middle" font-size="11" font-weight="600" fill="#993c1d">Deduction Codes</text><text x="270" y="${row2Y + 94}" text-anchor="middle" font-size="10" fill="#d85a30">Recoveries</text>`;
  svg += `</svg>`;
  wrap.innerHTML = svg;
}

function showFlowTip(evt, el) {
  const raw = decodeURIComponent(el.dataset.ed);
  const data = JSON.parse(raw);
  const tip = document.getElementById('tooltip');
  let html = `<h5>${data.dept}</h5>`;
  if (data.codes.length===0) html += '<div class="ed-item"><span class="ed-desc">No specific codes mapped</span></div>';
  else { data.codes.forEach(c=>{ html+=`<div class="ed-item"><span class="ed-code">${c.code}</span><span class="ed-desc" style="margin-left:6px;">${c.long.length>45?c.long.slice(0,45)+'…':c.long}</span></div>`; }); if((deptEdMap[data.dept]||[]).length>6) html+=`<div style="font-size:11px;color:#9a9890;margin-top:4px;">+ more codes — click department</div>`; }
  tip.innerHTML = html; tip.style.display = 'block'; positionTip(evt);
}

function hideTip(){ document.getElementById('tooltip').style.display='none'; }
